다음은 패널(Panel) 생성자 함수(Constructor Function) 입니다.


function Panel(element, canClose, closeHandler) {
  this.element = element;
  this.canClose = canClose;
  this.closeHandler = function () { if (closeHandler) closeHandler() };
}
